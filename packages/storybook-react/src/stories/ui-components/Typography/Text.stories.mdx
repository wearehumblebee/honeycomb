import { ArgsTable, Canvas, Meta, Story } from '@storybook/addon-docs';
import { select, text } from '@storybook/addon-knobs';
import { Text, useThemeUI } from 'theme-ui';
import { Paragraph } from '@humblebee/ui-react';

# Text

Text is a component rendering styled text blocks.

https://theme-ui.com/components/text

<Meta
  title="UI Components / Typography / Text"
  component={Text}
  parameters={{ layout: 'centered' }}
/>

<ArgsTable of={Text} />

It default to an html `<span />` element but you can customise that using the `as` prop from emotion

<Canvas>
  <Story name="Default">
    {() => {
      const { theme } = useThemeUI();
      return (
        <Text
          as="p"
          variant={select('Variant', theme ? Object.keys(theme.text || {}) : [], 'body')}
        >
          {text('content', 'Lorem ipsum dolor sit amet, consectetur adipiscing elit.')}
        </Text>
      );
    }}
  </Story>
</Canvas>

Or use Humblebee's Paragraph component which is basically a Text defaulting to an HTML paragraph element
